[pi4]
dtoverlay=vc4-fkms-v3d
max_framebuffers=2
arm_64bit=1
arm_boost=1

[all]

# Enable audio (loads snd_bcm2835)
dtparam=audio=on

# Enable I2C interfaces on the GPIO header
dtparam=i2c_arm=on

# Enable SPI for SJ201 Flashing
dtparam=spi=on

#enable BT
dtparam=krnbt=on

# Enable Camera Support
start_x=1

# Set 'force_turbo=1' to disable dynamic overclocking and enable overclocking always
force_turbo=1

# have a properly sized image
disable_overscan=1

# for sound over HDMI
# hdmi_drive=2
hdmi_ignore_cec_init=1


# Spec GPU Memory
gpu_mem=256
gpu_mem_256=128
gpu_mem_512=128
gpu_mem_1024=128

# Disable rainbow splash screen
disable_splash=1

# initramfs
enable_uart=1
upstream_kernel=1
kernel=kernel8.img
initramfs initramfs followkernel
